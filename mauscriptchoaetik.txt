local Players = game:GetService("Players")
local Animals = workspace:WaitForChild("Animals")

-- 💰 Leaderstats
Players.PlayerAdded:Connect(function(player)
	local stats = Instance.new("Folder")
	stats.Name = "leaderstats"
	stats.Parent = player

	local Money = Instance.new("IntValue")
	Money.Name = "Money"
	Money.Value = 0
	Money.Parent = stats
end)

-- 🐶 Làm động vật bị bệnh
local function MakeSick(animal)
	local humanoid = animal:FindFirstChild("Humanoid")
	if humanoid then
		humanoid.Health = humanoid.MaxHealth * 0.5
	end
end

for _, animal in pairs(Animals:GetChildren()) do
	MakeSick(animal)
end

Animals.ChildAdded:Connect(function(animal)
	task.wait(1)
	MakeSick(animal)
end)

-- 💊 HealPad hệ thống (tự tìm HealPad trong Workspace)
local HealPad = workspace:WaitForChild("HealPad")

HealPad.Touched:Connect(function(hit)
	local model = hit.Parent
	local humanoid = model:FindFirstChild("Humanoid")

	if humanoid then
		-- hồi máu
		humanoid.Health = math.min(humanoid.Health + 30, humanoid.MaxHealth)

		-- thưởng tiền nếu là người chơi
		local player = Players:GetPlayerFromCharacter(model)
		if player and player:FindFirstChild("leaderstats") then
			player.leaderstats.Money.Value += 10
		end
	end
end)
